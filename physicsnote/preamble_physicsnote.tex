%%  Format of enumerate environment
\renewcommand{\labelenumi}{(\arabic{enumi})}

%%  Numbering
\counterwithin{figure}{section}
\counterwithin{table}{section}
\numberwithin{equation}{section}

%%  Format of footnote
\renewcommand{\thefootnote}{*\arabic{footnote}}

%%  Theorem environment
\theoremstyle{definition}
\newtheorem{axm}{公理}
\newtheorem{define}{定義}
\newtheorem{mathdefine}{数学の定義}
\newtheorem{prop}{命題}
\newenvironment{axiom}{\begin{tcolorbox}[sharp corners, colback=blue!10, colframe=white, breakable=true]\begin{axm}}{\end{axm}\end{tcolorbox}}
\newenvironment{definition}{\begin{tcolorbox}[sharp corners, colback=blue!10, colframe=white, breakable=true]\begin{define}}{\end{define}\end{tcolorbox}}
\newenvironment{mathdefinition}{\begin{tcolorbox}[sharp corners, colback=blue!10, colframe=white, breakable=true]\begin{mathdefine}}{\end{mathdefine}\end{tcolorbox}}
\newenvironment{proposition}{\begin{tcolorbox}[sharp corners, colback=red!10, colframe=white, breakable=true]\begin{prop}}{\end{prop}\end{tcolorbox}}

%%  Proof environment
\renewcommand{\proofname}{\textgt{証明}}
\renewcommand{\qedsymbol}{$\blacksquare$}
